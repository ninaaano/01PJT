1.  개발부서의 김사장입니다       형식조회  (오름차순으로 정렬)

SELECT
dept_name || '서의 ' || name || '입니다' AS "부서의 이름입니다."
FROM emp_test
ORDER BY dept_name;
// 쓰레기 값이 나올 수 있기 때문에 뒤에 , name;을 붙여준다
// ORDER BY 1

2. 부서별/연봉이 높은 순서 이름/부서/연봉 조회

SELECT
name, dept_name,salary*12
FROM emp_test
ORDER BY dept_name DESC, salary DESC;

3. 연봉이 높은순서,  입사연도가 빠른순서 연봉/입사년도/이름 조회

SELECT
salary*12,in_date,name
FROM emp_test
ORDER BY salary DESC, in_date ASC;
